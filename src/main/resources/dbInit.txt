DROP TABLE USER_TYPES;
DROP TABLE USER;
DROP TABLE LOGIN;
DROP TABLE ASSIGMENT;
DROP TABLE STUDENT_ASSIGMENT;
CREATE TABLE USER_TYPES(TYPE_ID INTEGER NOT NULL UNIQUE PRIMARY KEY, TYPE TEXT UNIQUE NOT NULL);
CREATE TABLE USER(USER_ID INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,NAME TEXT NOT NULL UNIQUE, TYPE_ID INTEGER NOT NULL, FOREIGN KEY('TYPE_ID') REFERENCES USER_TYPES ('TYPE_ID'));
CREATE TABLE LOGIN(LOGIN TEXT UNIQUE NOT NULL, PASSWORD TEXT NOT NULL, USER_ID INTEGER UNIQUE NOT NULL, FOREIGN KEY('USER_ID') REFERENCES USER('USER_ID'));
CREATE TABLE ASSIGMENT(ASSIGMENT_ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, ASSIGMENT_NAME TEXT NOT NULL);
CREATE TABLE STUDENT_ASSIGMENT(STUDENT_ID INTEGER NOT NULL UNIQUE , ASSIGMENT_ID INTEGER NOT NULl UNIQUE, SUBMIT TEXT, GRADE INTEGER);
